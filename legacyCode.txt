    // ... (Your existing dice-building loop remains the same) ...

    const rollPromise = new Promise((resolve) => {
        playerOrEnemy.Box.onRollComplete = (rollResult) => {
            for (let i = 0; i < rollResult.length; i++) {
                playerOrEnemy.value += rollResult[i].value;
            }
            this.rolling = false;
            resolve(); // 2. This "unblocks" the await below

    });

    // 3. Trigger the roll
    playerOrEnemy.Box.roll(diceToRoll);

    // 4. Wait here until the promise is resolved

}
